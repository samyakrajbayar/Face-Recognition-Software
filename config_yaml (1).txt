# Face Recognition System Configuration
# =====================================

# Recognition settings
recognition:
  tolerance: 0.6              # Lower = more strict, higher = more lenient (0.1-1.0)
  model: 'hog'                # 'hog' for CPU, 'cnn' for GPU (more accurate but slower)
  max_faces_per_frame: 10     # Maximum number of faces to process per frame

  path: 'faces.db'            # Path to database file
  backup_enabled: true        # Enable automatic backups
  backup_interval: 86400      # Backup interval in seconds (24 hours)
  max_backup_files: 5         # Maximum number of backup files to keep

# Video capture settings
video:

  frame_width: 640            # Video frame width
  frame_height: 480           # Video frame height
  fps: 30                     # Target FPS
  buffer_size: 1              # Camera buffer size (smaller = less delay)
  display_fps: true           # Show FPS counter on video

# Performance optimization

  process_every_n_frames: 1   # Process every N frames (higher = faster but less accurate)
  resize_factor: 0.25         # Resize factor for processing (smaller = faster)
  use_gpu: false              # Use GPU acceleration (requires CUDA and dlib with GPU)
  max_threads: 4              # Maximum number of processing threads
  enable_optimization: true   # Enable various performance optimizations

# Attendance logging

    # - 'temperature'
